  8 var download = {
  9 		/**
 10 		 * Creates list with with persistent ids and modifications for all
 11 		 * Gears entites in <code>gears_entities</code> array.
 12 		 * 
 13 		 * @returns	{Array}	list with objects holding persistent id and
 14 		 * 					modifications
 15 		 */
 16 		idsList: function(gears_entities) {
 17 			var list = new Array();
 18 			
 19 			for (var i in gears_entities) {
 20 				list.push({
 21 					id: gears_entities[i].persistentId,
 22 					modifications: gears_entities[i].modifications
 23 				});
 24 			}
 25 			
 26 			return list;
 27 		},

 29 		 * @namespace Holds all methods related to downloading surveys updates

 35 			 * Gears surveys objects.
 36 			 * 
 37 			 * @returns	{Array}	list with objects holding persistent id and
 38 			 * 					modifications
 39 			 */

 41 				var gears_surveys = gears.survey.findAll();

 49 						download.survey.create(response.created);
 50 						download.survey.remove(response.removed);
 51 						download.survey.update(response.updated);
 52 
 53 						return response;
 54 					},
 55 					error: function(response, ioArgs) {
 56 						return response;
 57 					},
 58 					handleAs: "json"
 59 				});
 60 			},

 90 				}

206 gears.answer.createFromJSON = function(jsonAnswer, gears_question) {
207 	// store answer from jsonAnswer in local DB
208 	var gears_answer = new gears.Answer({
209 		persistentId: jsonAnswer.id,
210 		modifications: jsonAnswer.modifications,
211 		position: jsonAnswer.position,
212 		isText: jsonAnswer.isText,
213 		textValue: jsonAnswer.textValue,
214 		numericValue: jsonAnswer.numericValue,
215 		Question: gears_question
216 	});
217 	gears_answer.save();
218 };

220 gears.question.createFromJSON = function(jsonQuestion, gears_component) {
221 	// store question from jsonQuestion in local DB
222 	var gears_question = new gears.Question({
223 		persistentId: jsonQuestion.id,
224 		modifications: jsonQuestion.modifications,
225 		type: jsonQuestion.type,
226 		kind: jsonQuestion.kind,
227 		content: jsonQuestion.content,
228 		Component: gears_component
229 	});
230 	gears_question.save();
231 
232 	var jsonAnswers = jsonQuestion.answers;
233 	
234 	// store answers associated with jsonQuestion in local DB
235 	for (var i in jsonAnswers)
236 		gears.answer.createFromJSON(jsonAnswers[i], gears_question);
237 };

239 gears.component.createFromJSON = function(jsonComponent, gears_parent) {
240 	// store component from jsonComponent in local DB
241 	var gears_component = new gears.Component({
242 		persistentId: jsonComponent.id,
243 		modifications: jsonComponent.modifications,
244 		position: jsonComponent.position,
245 		Component: gears_parent
246 	});
247 	gears_component.save();
248 
249 	// store question object associated with jsonComponent in local DB
250 	if (jsonComponent.question)
251 		gears.question.createFromJSON(jsonComponent.question, gears_component);
252 	
253 	var jsonComponents = jsonComponent.components;
254 	
255 	// store all children components of jsonComponent in local DB 
256 	for (var i in jsonComponents)
257 		gears.component.createFromJSON(jsonComponents[i], gears_component);
258 };

260 gears.survey.createFromJSON = function(jsonSurvey) {
261 	var jsonRootComponent = jsonSurvey.rootComponent;
262 	
263 	var rootComponent = new Component(
264 			null,
265 			jsonRootComponent.modifications,
266 			jsonRootComponent.position,
267 			null,
268 			jsonRootComponent.id);
269 	
270 	var survey = new Survey(
271 			jsonSurvey.name,
272 			jsonSurvey.modifications,
273 			jsonSurvey.description,
274 			jsonSurvey.expirationDate,
275 			rootComponent,
276 			jsonSurvey.id);
277 	
278 	// creates survey and rootComponent in local DB
279 	var gears_survey = gears.survey.create(survey);
280 	
281 	var jsonComponents = jsonRootComponent.components;
282 	
283 	// store each component from jsonSurvey rootComponent in DB
284 	for (var i in jsonComponents) {
285 		gears.component.createFromJSON(jsonComponents[i],
286 				gears_survey.Component);
287 	}
288 };
