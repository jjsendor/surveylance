1 /**
  2  * @namespace Answer Gears DAO.
  3  */
  4 gears.answer = {
  5 		create: function(answer) {
  6 			var question = answer.question;
  7 			var gears_question = new gears.Question(question.gearsId);
  8 		
  9 			if (answer.position)
 10 				adjustAnswers(gears_question, answer.position);
 11 			
 12 			var gears_answer = new gears.Answer({
 13 				persistentId: answer.persistentId,
 14 				modifications: answer.modifications,
 15 				position: answer.position,
 16 				isText: (question.type == "text" ? 1 : 0),
 17 				textValue: (question.type == "text" && answer.value ? answer.value.toString() : null),
 18 				numericValue: (question.type == "numeric" && answer.value ? parseInt(answer.value) : null),
 19 				Question: gears_question
 20 			});
 21 			
 22 			gears_answer.save();
 23 			
 24 			answer.gearsId = gears_answer.rowid;
 25 			console.debug("Gears ORM: Answer created (persistentId: "
 26 					+ answer.persistentId + ", gearsId: " + answer.gearsId + ")");
 27 			
 28 			return gears_answer;
 29 		},
 30 		remove: function(gears_answer) {
 31 			console.debug("Gears ORM: Deleting answer (gearsId: "
 32 					+ gears_answer.rowid + ")");
 33 			if (gears_answer.position) 
 34 				readjustAnswers(gears_answer.Question, gears_answer.position);
 35 			
 36 			gears_answer.Question = null;
 37 			gears_answer.remove();
 38 		},
 39 		removeById: function(id) {
 40 			var gears_answer = new gears.Answer(id);
 41 
 42 			if (gears_answer)
 43 				gears.answer.remove(gears_answer);
 44 		},
 45 		update: function(answer) {
 46 			console.debug("Gears ORM: Updating answer (gearsId: "
 47 					+ answer.gearsId + ")");
 48 			var gears_answer = new gears.Answer(answer.gearsId);
 49 			
 50 			console.debug("Gears ORM: answer's question type = "
 51 					+ gears_answer.Question.type);
 52 			
 53 			if (answer.value) {
 54 				if (gears_answer.Question.type == "text")
 55 					gears_answer.textValue = answer.value.toString();
 56 				else if (gears_answer.Question.type == "numeric")
 57 					gears_answer.numericValue = parseInt(answer.value);
 58 			}
 59 			
 60 			gears_answer.save();
 61 		},
 62 		findAll: function() {
 63 			return gears.Answer.all().toArray();
 64 		},
 65 		findById: function(id) {
 66 			return new gears.Answer(id);
 67 		},
 68 		findByPersistentId: function(persistentId) {
 69 			return gears.Answer.filter(
 70 					"Answer.persistentId = ?", persistentId).first();
 71 		},
 72 		findByQuestion: function(gears_question) {
 73 			return gears.Answer.filter("Answer.Question = ?", gears_question.rowid)
 74 			.toArray();
 75 		}
 76 }
