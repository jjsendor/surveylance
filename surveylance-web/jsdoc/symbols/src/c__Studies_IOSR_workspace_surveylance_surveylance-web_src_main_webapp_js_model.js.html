 40 /**
 41  * Component helper class.
 42  *
 43  * @param	id
 44  * @param	modifications
 45  * @param	position
 46  * @param	parent
 47  * @param	persistentId
 48  */
 49 function Component(id, modifications, position, parent, persistentId) {
 50 	if (id === undefined)
 51 		id = null;
 52 	
 53 	if (modifications === undefined)
 54 		modifications = null;
 55 	
 56 	if (position === undefined)
 57 		position = null;
 58 	
 59 	if (parent === undefined)
 60 		parent = null;
 61 	
 62 	if (persistentId === undefined)
 63 		persistentId = null;
 64 	
 65 	this.id = id;
 66 	this.persistentId = persistentId;
 67 	this.gearsId = null;
 68 
 69 	this.modifications = modifications;
 70 	this.position = position;
 71 	this.parent = parent;
 72 	this.childs = new Array();
 73 	this.question = null;
 74 	this.dndSource = null;
 75 	
 76 	this.findChildById = function(childId) {
 77 		var components = dojo.filter(this.childs, function (item) {
 78 			return item.id == childId;
 79 		});
 80 		
 81 		if (components.length == 0)
 82 			return null;
 83 		
 84 		return components.pop();
 85 	};
 86 	
 87 	this.addChild = function(child) {
 88 		this.childs.push(child);
 89 		child.parent = this;
 90 	};
 91 	
 92 	this.removeChild = function(child) {
 93 		this.childs = dojo.filter(this.childs, function (item) {
 94 			return item != child;
 95 		});
 96 		child.parent = null;
 97 	};
 98 }
