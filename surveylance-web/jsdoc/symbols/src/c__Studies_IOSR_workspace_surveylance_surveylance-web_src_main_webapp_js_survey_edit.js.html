 30 function wipe(id) {
 31 	var animArgs = {
 32 		node: id,
 33 		duration: 500 // ms to run animation
 34 	};
 35 	
 36 	if (wipedOut[id])
 37 		dojo.fx.wipeIn(animArgs).play();
 38 	else
 39 		dojo.fx.wipeOut(animArgs).play();
 40 
 41 	wipedOut[id] = !wipedOut[id];
 42 }

 65 function gears_restoreComponent(gears_component) {
 66 	console.debug("Gears: restoring component with gearsId = "
 67 			+ gears_component.rowid);
 68 	console.debug("gears_component.Component.rowid = " + gears_component.Component.rowid);
 69 	if (gears_component.Component.rowid) {
 70 		parent = components[gears_component.Component.rowid];
 71 		console.debug("gears.question.findByComponent(" + gears_component + ")");
 72 		var gears_question = gears.question.findByComponent(gears_component);
 73 		console.debug("gears_question = " + gears_question);
 74 		if (gears_question) {
 75 			console.debug("Creating component with id " + gears_component.rowid
 76 					+ " for question with id " + gears_question.rowid
 77 					+ ", parent id: " + gears_question.Component.rowid);
 78 			type = ["question", gears_question.type, gears_question.kind];
 79 			
 80 			parent.dndSource.insertNodes(false, [{
 81 				id: gears_component.rowid,
 82 				type: type,
 83 				content: gears_question.content
 84 			}]);
 85 			
 86 			question = addComponent(type, parent.dndSource);
 87 			question.gearsId = gears_question.rowid;
 88 			question.parent.gearsId = gears_component.rowid;
 89 			
 90 			question.content = gears_question.content;
 91 			
 92 			//restoring answers
 93 			var gears_answers = gears.answer.findByQuestion(gears_question);
 94 			
 95 			for (var i = 0; i < gears_answers.length; i++) {
 96 				var gears_answer = gears_answers[i];
 97 				var answer = new Answer(
 98 					gears_answer.modifications,
 99 					gears_answer.position,
100 					gears_answer.isText
101 							? gears_answer.textValue
102 							: gears_answer.numericValue
103 				);
104 				
105 				answer.gearsId = gears_answer.rowid;
106 				
107 				addAnswer(question, answer);
108 			}
109 			
110 			return;	// do not iterate through child components
111 		}
112 		else {
113 			console.debug("Creating component with id "
114 					+ gears_component.rowid + ", parent id: "
115 					+ gears_component.Component.rowid);
116 			type = ["component", "order"];
117 			parent.dndSource.insertNodes(false, [{
118 				id: gears_component.rowid,
119 				type: type,
120 				position: gears_component.position
121 			}]);
122 			
123 			component = addComponent(type, parent.dndSource);
124 			component.gearsId = gears_component.rowid;
125 			
126 			components[component.gearsId] = component;
127 		}
128 	}
129 	
130 	var gears_components = gears.component.findByComponent(gears_component).sort(
131 			gears_sortComponents);
132 	console.debug("Restoring child components [length = "
133 			+ gears_components.length + "]");
134 	for (var i = 0; i < gears_components.length; i++)
135 		gears_restoreComponent(gears_components[i]);
136 }

590 		function(action, actionLog){
591 			switch (action.name) {
592 			case "create component":
593 				var component = action.data;
594 				var parent = component.parent;
595 
596 				if (!parent) {
597 					console.log("Replaying: create component aborted (no parent)");
598 					// quit current replay action and continue replaying
599 					actionLog.continueReplay();
600 					return;
601 				}
602 
603 				if (!parent.persistentId) {
604 					var gears_parent = new gears.Component(parent.gearsId);
605 
606 					if (!gears_parent.persistentId) {
607 						// quit current replay action and continue replaying
608 						actionLog.continueReplay();
609 						return;
610 					}
611 
612 					parent.persistentId = gears_parent.persistentId;
613 				}
614 
615 				xhr.component.create(component, actionLog);
616 				break;
617 			case "delete component":
618 				var component = action.data;
619 
620 				if (!component.persistentId) {
621 					var gears_component = new gears.Component(component.gearsId);
622 
623 					if (!gears_component.persistentId) {
624 						// quit current replay action and continue replaying
625 						actionLog.continueReplay();
626 						return;
627 					}
628 
629 					component.persistentId = gears_component.persistentId;
630 				}
631 				
632 				xhr.component.remove(component, actionLog);
633 				break;
634 			case "move component":
635 				var component = action.data;
636 
637 				if (!component.persistentId) {
638 					var gears_component = new gears.Component(component.gearsId);
639 
640 					if (!gears_component.persistentId) {
641 						// quit current replay action and continue replaying
642 						actionLog.continueReplay();
643 						return;
644 					}
645 
646 					component.persistentId = gears_component.persistentId;
647 				}
648 
649 				xhr.component.move(component, actionLog);
650 				break;
651 			case "create question":
652 				var question = action.data;
653 				var component = question.parent;
654 				var parent = component.parent;
655 
656 				if (!parent) {
657 					// quit current replay action and continue replaying
658 					actionLog.continueReplay();
659 					return;
660 				}
661 
662 				if (!parent.persistentId) {
663 					var gears_parent = new gears.Component(parent.gearsId);
664 
665 					if (!gears_parent.persistentId) {
666 						// quit current replay action and continue replaying
667 						actionLog.continueReplay();
668 						return;
669 					}
670 
671 					parent.persistentId = gears_parent.persistentId;
672 				}
673 				
674 				xhr.question.create(question, actionLog);
675 				break;
676 			case "delete question":
677 				var question = action.data;
