 10 function answerToJSON(answer) {
 11 	return {
 12 		persistentId: answer.persistentId,
 13 		gearsId: answer.gearsId,
 14 		modifications: answer.modifications,
 15 		position: answer.position,
 16 		value: answer.value
 17 	};
 18 }
 19 
 20 function questionToJSON(question) {
 21 	var answers = question.answers;
 22 	var jsonAnswers = new Array();
 23 
 24 	for (var i = 0; i < answers.length; i++) {
 25 		var answer = answers[i];
 26 		var jsonAnswer = answerToJSON(answer);
 27 		jsonAnswers.push(jsonAnswer);
 28 	}
 29 
 30 	return {
 31 		persistentId: question.persistentId,
 32 		gearsId: question.gearsId,
 33 		modifications: question.modifications,
 34 		content: question.content,
 35 		type: question.type,
 36 		kind: question.kind,
 37 		answers: jsonAnswers
 38 	};
 39 }

 41 function componentToJSON(component) {
 42 	var question = component.question;
 43 
 44 	var jsonQuestion = null;
 45 	var jsonComponents = null;
 46 
 47 	if (question != null) {
 48 		jsonQuestion = questionToJSON(question);
 49 	}
 50 	else {
 51 		var components = component.childs;
 52 		jsonComponents = new Array();
 53 
 54 		for (var i = 0; i < components.length; i++) {
 55 			var child = components[i];
 56 			var jsonChildComponent = componentToJSON(child);
 57 			jsonComponents.push(jsonChildComponent);
 58 		}
 59 	}
 60 
 61 	return {
 62 		persistentId: component.persistentId,
 63 		gearsId: component.gearsId,
 64 		modifications: component.modifications,
 65 		position: component.position,
 66 		components: jsonComponents,
 67 		question: jsonQuestion
 68 	};
 69 }

 71 /**

 76 function surveyToJSON(survey) {
 77 	return {
 78 		persistentId: survey.persistentId,
 79 		gearsId: survey.gearsId,
 80 		modifications: survey.modifications,
 81 		name: survey.name,
 82 		description: survey.description,
 83 		expirationDate: survey.expirationDate,
 84 		rootComponent: componentToJSON(survey.rootComponent)
 85 	};
 86 }

 88 function restoreAnswerPersistentId(jsonAnswer) {
 89 	var gears_answer = new gears.Answer(jsonAnswer.gearsId);
 90 	gears_answer.persistentId = jsonAnswer.persistentId;
 91 	gears_answer.save();
 92 }

 94 function restoreQuestionPersistentId(jsonQuestion) {
 95 	var jsonAnswers = jsonQuestion.answers;
 96 
 97 	for (var i = 0; i < jsonAnswers.length; i++) {
 98 		var jsonAnswer = jsonAnswers[i];
 99 		restoreAnswerPersistentId(jsonAnswer);
100 	}
101 
102 	var gears_question = new gears.Question(jsonQuestion.gearsId);
103 	gears_question.persistentId = jsonQuestion.persistentId;
104 	gears_question.save();
105 }
106 
107 function restoreComponentPersistentId(jsonComponent) {
108 	var question = component.question;
109 
110 	var jsonQuestion = jsonComponent.question;
111 	var jsonComponents = jsonComponent.components;
112 
113 	if (jsonQuestion != null) {
114 		restoreQuestionPersistentId(jsonQuestion);
115 	}
116 	else {
117 		for (var i = 0; i < jsonComponents.length; i++) {
118 			var jsonChildComponent = jsonComponents[i];
119 			restoreComponentPersistentId(jsonChildComponent);
120 		}
121 	}
122 
123 	var gears_component = new gears.Component(jsonComponent.gearsId);
124 	gears_component.persistentId = jsonComponent.persistentId;
125 	gears_component.save();
126 }

128 function restoreSurveyPersistentId(jsonSurvey) {
129 	restoreComponentPersistentId(jsonSurvey.rootComponent);
130 
131 	var gears_survey = new gears.Survey(jsonSurvey.gearsId);
132 	gears_survey.persistentId = jsonSurvey.persistentId;
133 	gears_survey.save();
134 }

136 /**
137  * Calls synchronization controller by XHR with POST parameter "survey"
138  * which is string with survey structure in JSON.
139  */
140 function gears_synchronizeSurvey() {
141 	var jsonSurvey = surveyToJSON(survey);
142 	console.debug("Survey in JSON format:\n" + dojo.toJson(jsonSurvey));
143 
144 	dojo.xhrPost({
145 		url: "../../survey/synchronize/" + survey.persistentId,
146 		content: {
147 			survey: dojo.toJson(jsonSurvey)
148 		},
149 		load: function(response, ioArgs) {
150 			console.debug("synchronized: " + response.synchronized);
151 
152 			if (response.synchronized) {
153 				console.debug(dojo.toJson(response.survey));
154 				restoreSurveyPersistentId(response.survey);
155 				new MessageBox(SYNCHRONIZATION_COMPLETED).show();
156 			}
157 			else
158 				new MessageBox(SYNCHRONIZATION_ERROR, "error").show();
159 
160 			return response;
161 		},
162 		error: function(response, ioArgs) {
163 			console.debug("ERROR: HTTP status code is ", ioArgs.xhr.status);
164 
165 			if (ioArgs.xhr.status != 200)
166 				new MessageBox(SYNCHRONIZATION_ERROR, "error").show();
167 
168 			return response;
169 		},
170 		handleAs: "json"
171 	});
172 }

179 gears.answer.restoreFromJSON = function(jsonAnswer, gears_question) {
180 	// Save jsonAnswer to DB
181 	var gears_answer = new gears.Answer({
182 		persistentId: jsonAnswer["persistentId"],
183 		modifications: jsonAnswer["modifications"],
184 		position: jsonAnswer["position"],
185 		isText: jsonAnswer["isText"],
186 		textValue: jsonAnswer["textValue"],
187 		numericValue: jsonAnswer["numericValue"],
188 		Question: gears_question
189 	});
190 	gears_answer.save();
191 }

193 gears.question.restoreFromJSON = function(jsonQuestion, gears_component) {
194 	// Save jsonQuestion to DB.
195 	var gears_question = new gears.Question({
196 		persistentId: jsonQuestion["persistentId"],
197 		modifications: jsonQuestion["modifications"],
198 		type: jsonQuestion["type"],
199 		kind: jsonQuestion["kind"],
200 		content: jsonQuestion["content"],
201 		Component: gears_component
202 	});
203 	gears_question.save();
204 
205 	// Restore answers associated with jsonQuestion to DB.
206 	for (var i = 0; i < jsonQuestion["answers"].length; i++) {
207 		var jsonAnswer = jsonQuestion["answers"][i];
208 		gears.answer.restoreFromJSON(jsonAnswer, gears_question);
209 	}
210 }

212 gears.component.restoreFromJSON = function(jsonComponent, gears_parentComponent) {
213 	// Save jsonComponent to DB.
214 	var gears_component = new gears.Component({
215 		persistentId: jsonComponent["persistentId"],
216 		modifications: jsonComponent["modifications"],
217 		position: jsonComponent["position"],
218 		Component: gears_parentComponent
219 	});
220 	gears_component.save();
221 
222 	// Restore question object associated with jsonComponent object.
223 	if (jsonComponent["question"]) {
224 		gears.question.restoreFromJSON(jsonComponent["question"],
225 				gears_component);
226 	}
227 	
228 	// Restore children of jsonComponent component.
229 	for (var i = 0; i < jsonComponent["components"].length; i++) {
230 		var jsonChildComponent = jsonComponent["components"][i];
231 		gears.component.restoreFromJSON(jsonChildComponent, gears_component);
232 	}
233 	
234 	// Return ID of new survey's root component.
235 	return gears_component.rowid;
236 }

238 gears.survey.restoreFromJSON = function(jsonSurvey) {
239 	// Remove old survey object from DB.
240 	gears.component.removeById(survey.rootComponent.gearsId);
241 	
242 	// Restore jsonSurvey structure to DB.
243 	var rootComponentID = gears.component.restoreFromJSON(
244 			jsonSurvey["rootComponent"], null);
245 	
246 	// Update survey object in DB.
247 	var gears_survey = gears.survey.findByPersistentId(
248 			jsonSurvey["persistentId"]);
249 	gears_survey.Component = rootComponentID;
250 	gears_survey.save();
251 }

276 				new MessageBox(SYNCHRONIZATION_ERROR, "error").show();
