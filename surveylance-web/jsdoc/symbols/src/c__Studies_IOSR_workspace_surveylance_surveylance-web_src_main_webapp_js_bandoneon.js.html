1 /*
  2  * Bandoneon - another accordion-like effect
  3  */
  4 
  5 dojo.require("dojo.fx");
  6 
  7 function Bandoneon() {
  8 	this.addPane = function(titleNodeId, paneNodeId) {
  9 		var titleNode = dojo.byId(titleNodeId);
 10 		var paneNode = dojo.byId(paneNodeId);
 11 		
 12 		var pane = new BandoneonPane(titleNode, paneNode);
 13 		
 14 		dojo.connect(titleNode, "onclick", pane, "togglePane");
 15 	}
 16 }
 17 
 18 function BandoneonPane(titleNode, paneNode) {
 19 	this.titleNode = titleNode;
 20 	this.paneNode = paneNode;
 21 	
 22 	this.togglePane = function() {
 23 		if (paneNode.style.display == "none") {
 24 			dojo.fx.wipeIn({ node: paneNode.id, duration: 500 }).play();
 25 			dojo.addClass(titleNode, "active");
 26 		}
 27 		else {
 28 			dojo.fx.wipeOut({ node: paneNode.id, duration: 500 }).play();
 29 			dojo.removeClass(titleNode, "active");
 30 		}
 31 	}
 32 }
