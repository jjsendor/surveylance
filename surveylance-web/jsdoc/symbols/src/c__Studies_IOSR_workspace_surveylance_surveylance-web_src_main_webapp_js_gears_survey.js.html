1 /**
  2  * @namespace Survey Gears DAO.
  3  */
  4 gears.survey = {
  5 		create: function(survey) {
  6 			console.debug("Gears ORM: Creating survey");
  7 			var rootComponent = survey.rootComponent;
  8 			
  9 			var gears_rootComponent = new gears.Component({
 10 				persistentId: rootComponent.persistentId,
 11 				modifications: rootComponent.modifications,
 12 				position: rootComponent.position,
 13 				Component: rootComponent.parent
 14 			}).save();
 15 			
 16 			var gears_survey = new gears.Survey({
 17 				persistentId: survey.persistentId,
 18 				name: survey.name,
 19 				modifications: survey.modifications,
 20 				description: survey.description,
 21 				expirationDate: survey.expirationDate,
 22 				Component: gears_rootComponent
 23 			}).save();
 24 			
 25 			return gears_survey;
 26 		},
 27 		remove: function(gears_survey) {
 28 			console.debug("Gears ORM: Deleting survey (gearsId "
 29 					+ gears_survey.rowid + ")");
 30 			// delete rootComponent and all component underneath it recursively
 31 			gears.component.remove(gears_survey.rootComponent);
 32 			gears_survey.Component = null;
 33 			gears_survey.remove();
 34 		},
 35 		removeById: function(id) {
 36 			var gears_survey = new gears.Survey(id);
 37 
 38 			if (gears_survey)
 39 				gears.survey.remove(gears_survey);
 40 		},
 41 		findAll: function() {
 42 			return gears.Survey.all();
 43 		},
 44 		findById: function(id) {
 45 			return new gears.Survey(id);
 46 		},
 47 		findByPersistentId: function(persistentId) {
 48 			return gears.Survey.filter(
 49 					"Survey.persistentId = ?", persistentId).first();
 50 		}
 51 };
